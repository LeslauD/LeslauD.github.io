//压缩点云文件、删除原始深度图像数据、压缩分割后深度图像视频
void ImageProcess::compressData()
{
	unsigned long Len;
	if (subId == 3)
		Len = 424 * 512 * 3 * 150 * sizeof(FLOAT);
	else
		Len = 424 * 512 * 3 * sizeof(FLOAT);

	string cloudName = "./Data/Cloud_" + to_string(fileId) + "_" + to_string(subId) + ".dat";
	f2.open(cloudName, ios::binary);
	unsigned char* cloud = new unsigned char[Len];
	unsigned long cloudLen = Len;
	f2.read(reinterpret_cast<char*>(cloud), cloudLen);
	f2.close();
	unsigned char* buf = new unsigned char[Len];
	unsigned long bufLen = Len;
	compress(buf, &bufLen, cloud, cloudLen);
	f3.open(cloudName, ios::binary);
	f3.write(reinterpret_cast<char*>(buf), bufLen);
	f3.close();
	delete[] buf;
	delete[] cloud;
}
